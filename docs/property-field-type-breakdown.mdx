</ParamField>

<ParamField body="date" param-type="date" showLocation="false">

</ParamField>

</ParamField>

</ParamField>

</ParamField>

<ParamField body="integer" param-type="integer" showLocation="false">

</ParamField>

<ParamField body="keyword" param-type="keyword" showLocation="false">

</ParamField>

<ParamField body="nested" param-type="nested" showLocation="false">

</ParamField>

<ParamField body="taxonomy" param-type="taxonomy" showLocation="false">

<ParamField body="text" param-type="text" showLocation="false">

</ParamField>
